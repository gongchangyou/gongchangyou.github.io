template<typename _RandomAccessIterator>
void __reverse(_RandomAccessIterator __first, _RandomAccessIterator __last, random_access_iterator_tag)
{
	if (__first == __last)
		return;
	–__last;
	while (__first < __last)
	{
		std::iter_swap(__first, __last);
		++__first;
		–__last;
	}
}
